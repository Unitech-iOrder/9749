3 Controller (API 端點) 

[Route("api/[controller]")]
[ApiController]
public class EmployeesController : ControllerBase
{
    private readonly IEmployeeService _service;
    public EmployeesController(IEmployeeService service) => _service = service;

    [HttpGet]
    public IActionResult Get() => Ok(_service.GetAll());

    [HttpPost]
    public IActionResult Post(Employee emp) => Ok(_service.Create(emp));

    [HttpPut]
    public IActionResult Put(Employee emp) => _service.Update(emp) ? Ok("Updated") : NotFound();

    [HttpDelete("{id}")]
    public IActionResult Delete(int id) => _service.Delete(id) ? Ok("Deleted") : NotFound();
}

關鍵技術點

相依性注入 (DI)

在 Program.cs 務必註冊：
builder.Services.AddScoped<IEmployeeService, EmployeeService>();

日期格式處理

DateOfJoining 傳輸時應遵循 ISO 8601 格式，.NET 會自動解析為 DateTime2。
